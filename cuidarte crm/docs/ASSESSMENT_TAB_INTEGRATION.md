   try {
      const { error } = await supabase
         .from('clientes_ado_notion')
         .update({
            onboarding_phase2_completed: false,
            onboarding_phase2_completed_at: null
         })
         .eq('id', client.id);

      toast.error("Error al solicitar valoración");
   }
};
